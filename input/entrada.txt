int calcularFactorial(int n) {
    int fact;
    fact = 1;
    int j;
    
    for (j = 1; j <= n; j = j + 1) {
        fact = fact * j;
    }
    
    return fact;
}

int esPrimo(int num) {
    if (num <= 1) {
        return 0;
    }
    
    int k;
